Fibonacci (recursive, with accumulator)

function fibAcc (n) {
	var l, c = [0, 1];
	
	function f (n) {
		if (n < 2) {
			return c[c.length - 1];
		} else {
			l = c.length;
			c[l] = c[l - 1] + c[l - 2];
			return f(n - 1);
		}
	}
	
	return f(n);
}

Fibonacci (non-recursive)

function fibLoop (n) {
	var l, c = [0, 1];
	while (n >= c.length) {
		l = c.length;
		c[l] = c[l - 1] + c[l - 2];
	}
	return c[n];
}
